Example

$.getJSON('https://rappi.herokuapp.com/twitter/search.json?q=sunshine', function(result) {
  if (result.status !== 'ok') {
    alert(result.message);
  }
  // Add the tweets to the document.
  for (var i = 0; i < result.tweets.length; i++) {
    var p = document.createElement('p');
    p.innerHTML = result.tweets[i].text;
    document.body.appendChild(p);
  }
});
