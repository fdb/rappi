Example

  $.getJSON('https://rappi.herokuapp.com/meerkat/broadcasts.json', function(broadcasts) {
    // Show the first broadcast in an external window.
    var b = broadcasts[0];
    window.open(b.Url, '_blank', 'width=300,height=600');
  });
