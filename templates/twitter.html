Search for a phrase or keyword on Twitter and return a list of tweets.

Request

https://rappi.herokuapp.com/twitter/search.json?q=yoursearch

Response

...

Example

  $.getJSON('https://rappi.herokuapp.com/twitter/search.json?q=obama', function(tweets) {
    // Add the tweets to the document.
    for (var i = 0; i < tweets.length; i++) {
      var p = document.createElement('p');
      p.innerHTML = tweets[i].text;
      document.body.appendChild(p);
    }
  });
