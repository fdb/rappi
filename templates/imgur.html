Example

$.getJSON('https://rappi.herokuapp.com/imgur/search.json?q=flower', function(result) {
  if (result.status !== 'ok') {
    alert(result.message);
  }
  // Add the photos to the document.
  for (var i = 0; i < result.photos.length; i++) {
    var img = document.createElement('img');
    img.src = result.photos[i].imageUrl;
    img.width = 300;
    document.body.appendChild(img);
  }
});
